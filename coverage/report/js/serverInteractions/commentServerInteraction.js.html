function addComment(eventId, author, post_date, content, callback) {
    var authToken = LetsDoThis.Session.getInstance().getAuthToken();
	var commentUrl = "http://159.203.12.88/api/events/"+eventId+"/comments/";
	
    var postData = {
        "author": author,
        "post_date": post_date,
        "content": content
    }; 
    
	$.ajax({
		type: 'POST',
		url: commentUrl,
		dataType: 'json',
        contentType: 'application/json',
		beforeSend: function(xhr) {
				xhr.setRequestHeader("Authorization", "JWT " + authToken);
		},
        data: JSON.stringify(postData),
		success: function (resp) {
			console.log("comment added");
            callback(resp);
		},
		error: function(e) {
			console.log(e);
		}
	});
}
