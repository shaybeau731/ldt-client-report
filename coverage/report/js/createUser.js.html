$(document).ready(function(){
 
    $("#btn-submit").click(function(){
        $("#errorMessage").html("");
        var username = $('#username').val();
        var password = $('#password').val();
        var email = $('#email').val();
        var phone = $('#phone').val();
        
        checkIfInputIsValid(username, password, email, phone);
        
        addUser(username, password, email, phone);
    });
    
    console.log("Create user has loaded");
});
 
function checkIfInputIsValid(username, password, email, phone) {
    if (!username.match(/^[a-zA-Z\d]+$/)) {
        $("#errorMessage").append("<p>Username must consist of only letters and numbers</p>");
    }
    if (password.length == 0) {
        $("#errorMessage").append("<p>Password must be set</p>")
    }
    if (!phone.match(/^\d{10}$/)) {
        $("#errorMessage").append("<p>Phone number must be 10 numbers long (no dashes)");
    }
}
 
function addUser(username, password, email, phone) {
    createUser(username, password, email, phone, function(resp){
        window.location = "login.html";
    });
}
